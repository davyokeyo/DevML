model= tf.keras.models.Sequential()
#if the input to the layer has a rank greater than 2, then it is flattened prior
model.add(tf.keras.layers.Flatten())
#being a case of non-linear dataset we used an non-linear activation function: relu(rectified linear unit)
#unlike sigmoid and hyperbolic function activation ReLu is able to overcome the vanishing gradient problem
model.add(tf.keras.layers.Dense(150, activation=tf.nn.relu))
model.add(tf.keras.layers.Dense(150, activation=tf.nn.relu))
#
model.add(tf.keras.layers.Dense(10, activation=tf.nn.softmax))

model.compile(optimizer="adam",
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
model.fit(x_train,y_train, epochs=4)
